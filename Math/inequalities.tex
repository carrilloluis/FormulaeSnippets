\subsection*{Properties of Inequalities}
\begin{itemize}
\item{If $a < b$ then $a + c < b + c$ and $a - c < b - c$}
\item{If $a < b$ and $c > 0$ then $ac < bc$ and $a/c < b/c$}
\item{If $a < b$ and $c < 0$ then $ac > bc$ and $a/c > b/c$}
\end{itemize}
